## 📖 ما هي سيف؟

<div align="center">

```
┌─────────────────────────────────────────────────────┐
│                                                     │
│   .سيف  ←  حداد (مترجم)  ←  لغة الآلة (0 و 1)     │
│                                                     │
│   كود عربي  ←  ترجمة مباشرة  ←  برنامج تنفيذي      │
│                                                     │
└─────────────────────────────────────────────────────┘
```

</div>

**سيف** هي لغة برمجة عربية **مُجمّعة** (Compiled). ليست تغليفاً ولا ترجمة - بل مترجم حقيقي يحول الكود العربي مباشرة إلى لغة الآلة.

### مقارنة

| | Python | JavaScript | C | **سيف** |
|---|---|---|---|---|
| النوع | مفسرة | مفسرة | مجمعة | **مجمعة** |
| اللغة | إنجليزية | إنجليزية | إنجليزية | **عربية** |
| السرعة | 🐢 | 🐢 | 🚀 | 🚀 |
| المخرج | يحتاج مفسر | يحتاج متصفح | برنامج مستقل | **برنامج مستقل** |
| التعلم للعرب | متوسط | متوسط | صعب | **سهل** |

## ❓ لماذا سيف؟

### الواقع

> الحواسيب صُنعت بالإنجليزية. لغات البرمجة القوية (C, C++, Rust, Go) كلها بالإنجليزية.

### المحاولات السابقة

> لغات مثل "ألف" و "ج" و "لغة الضاد" هي **تغليف** - تترجم الكود العربي إلى إنجليزي ثم تترجمه. طبقة فوق طبقة.

### سيف: الحل الجذري

| الميزة | التفاصيل |
|--------|----------|
| 🔧 **مترجم معدل** | عدّلنا TCC من الداخل ليفهم العربية |
| 📝 **من الجذور** | التعديل في قلب المترجم، ليس تغليفاً |
| 🎯 **مباشر** | كود عربي ← لغة آلة (بدون وسيط) |
| 📚 **مكتبة عربية** | `سيف_خالص.ع` - كل الدوال الأساسية بالعربية |

## 📚 الدليل الكامل

### 1. هيكل البرنامج

```c
#ادرج "سيف_خالص.ع"      // المكتبة القياسية

عدد رئيسي() {             // الدالة الرئيسية - mandatory
    // كودك هنا
    ارجع نجاح;            // إنهاء ناجح
}
```

### 2. المتغيرات والثوابت

```c
// متغيرات
عدد س = 10;
كسري ك = 3.14;
مزدوج م = 2.718281;
محرف ح = 'أ';
طويل ط = 1000000;
قصير ق = 5;

// ثوابت
ثابت عدد سرعة_الضوء = 299792458;
```

### 3. الإدخال والإخراج

```c
// طباعة
اكتب("العدد: %d\n", س);
اكتب_سطر("سطر جديد تلقائي");

// قراءة
عدد عمر;
اقرأ("%d", &عمر);

محرف اسم[50];
اقرأ_سطر_عربي(اسم, 50);
```

### 4. الشروط

```c
اذا (عمر >= 18) {
    اكتب_سطر("بالغ");
} والا اذا (عمر >= 13) {
    اكتب_سطر("مراهق");
} والا {
    اكتب_سطر("طفل");
}
```

### 5. التكرار

```c
// طالما - while loop
عدد ع = 0;
طالما (ع < 5) {
    اكتب("%d\n", ع);
    ع = ع + 1;
}

// لكل - for loop
لكل (عدد ع = 0; ع < 5; ع = ع + 1) {
    اكتب("%d\n", ع);
}

// توقف - break
// استمر - continue
```

### 6. التفرع - Switch

```c
عدد خيار = 2;

تفرع (خيار) {
    حالة 1:
        اكتب_سطر("واحد");
        توقف;
    حالة 2:
        اكتب_سطر("اثنان");
        توقف;
    افتراضي:
        اكتب_سطر("غير معروف");
}
```

### 7. الدوال

```c
// تعريف دالة
عدد جمع(عدد أ, عدد ب) {
    ارجع أ + ب;
}

عدم تحية(محرف *اسم) {
    اكتب("مرحباً %s!\n", اسم);
}

// استخدام
عدد نتيجة = جمع(5, 3);
تحية("أرسلان");
```

### 8. المصفوفات

```c
عدد اعداد[5] = {1, 2, 3, 4, 5};
اعداد[0] = 10;

لكل (عدد ع = 0; ع < 5; ع = ع + 1) {
    اكتب("%d ", اعداد[ع]);
}
```

### 9. النصوص

```c
محرف اسم[50];
انسخ_نص(اسم, "أرسلان");           // strcpy
عدد طول = طول_نص(اسم);              // strlen

اذا (قارن_نص(اسم, "أرسلان") == 0) { // strcmp
    اكتب_سطر("متطابقان");
}
```

### 10. المؤشرات

```c
عدد س = 10;
عدد *مؤشر = &س;
*مؤشر = 20;    // س الآن = 20
```

### 11. الذاكرة الديناميكية

```c
عدد *مصفوفة = احجز_ذاكرة(10 * sizeof(عدد));
مصفوفة[0] = 100;
حرر_ذاكرة(مصفوفة);
```

### 12. الملفات

```c
// كتابة
ملف *م = افتح_ملف("بيانات.txt", وضع_كتابة);
اكتب_سطر_ملف("مرحباً", م);
اغلق_ملف(م);

// قراءة
محرف سطر[256];
م = افتح_ملف("بيانات.txt", وضع_قراءة);
طالما (!نهاية_ملف(م)) {
    اقرا_سطر(سطر, 256, م);
    اكتب_سطر(سطر);
}
اغلق_ملف(م);
```

### 13. الهياكل - Struct

```c
هيكل طالب {
    محرف اسم[50];
    عدد عمر;
};

هيكل طالب ط;
انسخ_نص(ط.اسم, "أرسلان");
ط.عمر = 25;
```

## 🎯 أمثلة

### آلة حاسبة

```c
#ادرج "سيف_خالص.ع"

عدد رئيسي() {
    عدد أ, ب;
    محرف ع;
    
    اكتب("العملية (+ - * /): "); اقرأ(" %c", &ع);
    اكتب("العدد الأول: "); اقرأ("%d", &أ);
    اكتب("العدد الثاني: "); اقرأ("%d", &ب);
    
    تفرع (ع) {
        حالة '+': اكتب("%d\n", أ + ب); توقف;
        حالة '-': اكتب("%d\n", أ - ب); توقف;
        حالة '*': اكتب("%d\n", أ * ب); توقف;
        حالة '/':
            اذا (ب != 0) اكتب("%d\n", أ / ب);
            والا اكتب_سطر("قسمة على صفر!");
            توقف;
        افتراضي: اكتب_سطر("غير معروف");
    }
    
    ارجع نجاح;
}
```

### جدول الضرب

```c
#ادرج "سيف_خالص.ع"

عدد رئيسي() {
    عدد ر;
    اكتب("أدخل رقماً: "); اقرأ("%d", &ر);
    
    لكل (عدد ع = 1; ع <= 10; ع = ع + 1) {
        اكتب("%d × %2d = %3d\n", ر, ع, ر * ع);
    }
    
    ارجع نجاح;
}
```

### تخمين رقم

```c
#ادرج "سيف_خالص.ع"
#ادرج <stdlib.h>
#ادرج <time.h>

عدد رئيسي() {
    srand(time(فارغ));
    عدد سري = rand() % 100 + 1;
    عدد ت, م = 0;
    
    اكتب_سطر("🎲 خمن الرقم (1-100):");
    
    طالما (صحيح) {
        اكتب("تخمينك: "); اقرأ("%d", &ت); م = م + 1;
        
        اذا (ت == سري) {
            اكتب("✅ صحيح! (%d محاولات)\n", م);
            توقف;
        } والا اذا (ت < سري) اكتب_سطر("📈 أكبر!");
        والا اكتب_سطر("📉 أصغر!");
    }
    
    ارجع نجاح;
}
```

## 🔧 كيف بنيت

### نظرة عامة

```
┌──────────────┐     ┌──────────────┐     ┌──────────────┐
│  TCC الأصلي  │ ──▶ │  تعديل 4     │ ──▶ │  مترجم حداد  │
│  (مترجم C)   │     │  ملفات       │     │  (يفهم سيف)  │
└──────────────┘     └──────────────┘     └──────────────┘
```

### الملفات المعدلة

| الملف | ماذا أضفنا |
|-------|------------|
| `tcctok.h` | 30+ كلمة عربية (`اذا`، `طالما`، `لكل`...) |
| `tccgen.c` | ربط الكلمات العربية بمنطق المترجم |
| `tccpp.c` | دعم `#ادرج` كبديل لـ `#include` |
| `libtcc.c` | امتدادات `.سيف` و `.ع` |

### تفاصيل التعديل

**`tcctok.h`** - إضافة الكلمات:
```c
DEF(TOK_IF, "if")
DEF(TOK_IF_AR, "اذا")         // ✨ كلمة عربية
DEF(TOK_WHILE, "while")
DEF(TOK_WHILE_AR, "طالما")    // ✨ كلمة عربية
DEF(TOK_RETURN, "return")
DEF(TOK_RETURN_AR, "أرجع")    // ✨ كلمة عربية
DEF(TOK_RETURN_AR2, "ارجع")   // ✨ بدون همزة
```

**`tccgen.c`** - ربط الكلمات:
```c
// قبل
if (t == TOK_IF) { ... }

// بعد
if (t == TOK_IF || t == TOK_IF_AR) { ... }
```

**`tccpp.c`** - توجيه `#ادرج`:
```c
case TOK_INCLUDE:
case TOK_INCLUDE_AR:    // #ادرج
    parse_include(...);
```

**`libtcc.c`** - امتدادات:
```c
|| !PATHCMP(ext, "سيف")  // .سيف
|| !PATHCMP(ext, "ع")    // .ع
```
